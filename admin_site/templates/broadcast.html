import asyncio import os import aiosqlite def send_broadcast(text_ru, text_en, text_ro, photo_url=None): """Потокобезопасный вызов рассылки из любого потока (например, из Flask)""" from bot import bot # импорт из bot/__init__.py from bot.main import dp # доступ к диспетчеру и его event loop async def _broadcast(): async with aiosqlite.connect(os.getenv("DB_PATH", "berry.db")) as db: cursor = await db.execute("SELECT user_id, language FROM users") users = await cursor.fetchall() for user_id, lang in users: try: if lang == 'ru': txt = text_ru elif lang == 'ro': txt = text_ro else: txt = text_en if photo_url: base_url = os.getenv("RENDER_EXTERNAL_URL", "").rstrip('/') full_url = base_url + photo_url await bot.send_photo(user_id, photo=full_url, caption=txt) else: await bot.send_message(user_id, txt) await asyncio.sleep(0.05) # чтобы не упереться в лимиты Telegram except Exception as e: print(f"Broadcast error for {user_id}: {e}") # Получаем event loop бота (он уже запущен в главном потоке) loop = dp.loop if hasattr(dp, 'loop') else asyncio.get_event_loop() # Потокобезопасно добавляем корутину в этот цикл asyncio.run_coroutine_threadsafe(_broadcast(), loop)
